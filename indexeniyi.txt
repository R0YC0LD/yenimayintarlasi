    <script>
        const SoundManager = {
            ctx: null,
            muted: false,
            init: function() {
                if (!this.ctx) {
                    this.ctx = new (window.AudioContext || window.webkitAudioContext)();
                }
                if (this.ctx.state === 'suspended') {
                    this.ctx.resume();
                }
            },
            toggleMute: function() {
                this.muted = !this.muted;
                return this.muted;
            },
            playTone: function(freq, type, duration, vol = 0.1) {
                if (this.muted || !this.ctx) return;
                const osc = this.ctx.createOscillator();
                const gain = this.ctx.createGain();
                osc.type = type;
                osc.frequency.setValueAtTime(freq, this.ctx.currentTime);
                gain.gain.setValueAtTime(vol, this.ctx.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.01, this.ctx.currentTime + duration);
                osc.connect(gain);
                gain.connect(this.ctx.destination);
                osc.start();
                osc.stop(this.ctx.currentTime + duration);
            },
            playNoise: function(duration, vol = 0.5) {
                if (this.muted || !this.ctx) return;
                const bufferSize = this.ctx.sampleRate * duration;
                const buffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                for (let i = 0; i < bufferSize; i++) {
                    data[i] = Math.random() * 2 - 1;
                }
                const noise = this.ctx.createBufferSource();
                noise.buffer = buffer;
                const gain = this.ctx.createGain();
                // Patlama zarfƒ± (Envelope)
                gain.gain.setValueAtTime(vol, this.ctx.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.01, this.ctx.currentTime + duration);
                
                // Lowpass filter ile patlamayƒ± tokla≈ütƒ±r
                const filter = this.ctx.createBiquadFilter();
                filter.type = "lowpass";
                filter.frequency.value = 800;

                noise.connect(filter);
                filter.connect(gain);
                gain.connect(this.ctx.destination);
                noise.start();
            },
            // Efektler
            sfxClick: function() { this.playTone(800, 'square', 0.05, 0.05); }, // Tƒ±k
            sfxDig: function() { this.playTone(300, 'triangle', 0.1, 0.1); }, // Kazma
            sfxFlag: function() { this.playTone(150, 'square', 0.1, 0.1); }, // Bayrak
            sfxSetMine: function() { this.playTone(100, 'sawtooth', 0.1, 0.1); }, // Mayƒ±n kurma
            sfxExplode: function() { this.playNoise(0.8, 0.8); }, // Patlama (G√ºr√ºlt√º)
            sfxWin: function() { // Zafer Melodisi
                if(this.muted) return;
                const now = this.ctx.currentTime;
                [523.25, 659.25, 783.99, 1046.50].forEach((freq, i) => { // C Major Arpeggio
                    const osc = this.ctx.createOscillator();
                    const gain = this.ctx.createGain();
                    osc.type = 'square';
                    osc.frequency.value = freq;
                    gain.gain.value = 0.1;
                    gain.gain.exponentialRampToValueAtTime(0.01, now + i*0.15 + 0.3);
                    osc.connect(gain);
                    gain.connect(this.ctx.destination);
                    osc.start(now + i*0.15);
                    osc.stop(now + i*0.15 + 0.3);
                });
            }
        };
    </script>

    <script>
        // --- Deƒüi≈ükenler ---
        let gridSize = 8;
        let totalCells = 64;
        let maxMines = 0;
        let mines = [];
        let gridState = []; // 0:kapalƒ±, 1:a√ßƒ±k, 2:bayrak
        let gameActive = false;
        let isSettingPhase = true;
        let currentMode = 'dig';
        let revealedCount = 0;
        let timer = 0;
        let timerInterval;

        // --- Ses Kontrol ---
        function toggleAudio() {
            const isMuted = SoundManager.toggleMute();
            document.getElementById('mute-btn').innerText = isMuted ? "üîá" : "üîä";
        }

        // --- Boyut Hesaplama (RESPONSIVE) ---
        function resizeGrid() {
            if (document.getElementById('screen-game').style.display !== 'flex') return;
            const headerHeight = 160; 
            const availableH = window.innerHeight - headerHeight;
            const availableW = window.innerWidth - 30; 
            let size = Math.min(availableH, availableW) / gridSize;
            size = Math.floor(size); 
            document.documentElement.style.setProperty('--cell-size', `${size}px`);
            const fontSize = Math.max(12, size * 0.6); 
            const cells = document.querySelectorAll('.cell');
            cells.forEach(c => c.style.fontSize = `${fontSize}px`);
        }

        window.addEventListener('resize', resizeGrid);

        // --- Ayarlar ---
        function selectSize(n) {
            SoundManager.init(); // Kullanƒ±cƒ± etkile≈üimiyle sesi ba≈ülat
            SoundManager.sfxClick();
            gridSize = n;
            totalCells = n * n;
            document.querySelectorAll('#screen-setup .win-btn').forEach(b => b.classList.remove('selected'));
            event.target.classList.add('selected');
        }

        // --- Faz 1: Ba≈ülat ---
        function startGame() {
            SoundManager.init();
            SoundManager.sfxClick();
            isSettingPhase = true;
            mines = [];
            maxMines = Math.floor(totalCells * 0.20);
            
            switchScreen('screen-game');
            document.getElementById('player-label').innerText = "Oyuncu 1: Tuzak Kurma";
            document.getElementById('p1-controls').style.display = 'flex';
            document.getElementById('p2-controls').style.display = 'none';
            document.getElementById('face-btn').innerText = "üôÇ";
            document.getElementById('max-lbl').innerText = maxMines;
            document.body.classList.remove('shake-screen'); // Reset shake

            updateCounter();
            createGrid(handleP1Click);
            resizeGrid(); 
        }

        function createGrid(clickHandler) {
            const grid = document.getElementById('game-grid');
            grid.innerHTML = '';
            grid.style.gridTemplateColumns = `repeat(${gridSize}, 1fr)`;
            
            for (let i = 0; i < totalCells; i++) {
                const cell = document.createElement('div');
                cell.classList.add('cell');
                cell.dataset.index = i;

                if (isSettingPhase && mines.includes(i)) {
                    cell.classList.add('has-mine-p1');
                }

                addMouseEffects(cell);
                cell.onclick = () => clickHandler(i, cell);
                cell.oncontextmenu = (e) => {
                    e.preventDefault();
                    if(!isSettingPhase) handleRightClick(i, cell);
                };
                grid.appendChild(cell);
            }
            resizeGrid();
        }

        function addMouseEffects(cell) {
            const setO = () => { if(gameActive && !isSettingPhase) document.getElementById('face-btn').innerText = "üòÆ"; };
            const setSmile = () => { if(gameActive && !isSettingPhase) document.getElementById('face-btn').innerText = "üôÇ"; };
            cell.addEventListener('mousedown', setO);
            cell.addEventListener('touchstart', setO, {passive: true});
            cell.addEventListener('mouseup', setSmile);
            cell.addEventListener('touchend', setSmile);
        }

        // --- P1 Mantƒ±ƒüƒ± ---
        function handleP1Click(index, cell) {
            const idx = mines.indexOf(index);
            if (idx > -1) {
                mines.splice(idx, 1);
                cell.classList.remove('has-mine-p1');
                SoundManager.sfxClick(); // Kaldƒ±rma sesi
            } else {
                if (mines.length < maxMines) {
                    mines.push(index);
                    cell.classList.add('has-mine-p1');
                    SoundManager.sfxSetMine(); // Mayƒ±n kurma sesi (Tok)
                }
            }
            updateCounter();
        }

        function finishSetting() {
            SoundManager.sfxClick();
            if (mines.length < 3) {
                alert("En az 3 mayƒ±n koymalƒ±sƒ±n!");
                return;
            }
            switchScreen('screen-handover');
        }

        // --- Faz 2: √á√∂zme Ba≈ülangƒ±cƒ± ---
        function startSolving() {
            SoundManager.sfxClick();
            isSettingPhase = false;
            gameActive = true;
            gridState = new Array(totalCells).fill(0);
            revealedCount = 0;
            currentMode = 'dig';
            document.body.classList.remove('shake-screen');

            timer = 0;
            clearInterval(timerInterval);
            timerInterval = setInterval(() => {
                timer++;
                if(timer<999) document.getElementById('timer-display').innerText = timer.toString().padStart(3,'0');
            }, 1000);

            switchScreen('screen-game');
            document.getElementById('player-label').innerText = "Oyuncu 2: √á√∂z√ºc√º";
            document.getElementById('p1-controls').style.display = 'none';
            document.getElementById('p2-controls').style.display = 'flex';
            document.getElementById('face-btn').innerText = "üôÇ";

            updateCounter();
            createGrid(handleP2Click);
        }

        // --- P2 Mantƒ±ƒüƒ± ---
        function handleP2Click(index, cell) {
            if (!gameActive || gridState[index] === 1) return;

            if (currentMode === 'flag') {
                toggleFlag(index, cell);
            } else {
                if (gridState[index] === 2) return; 
                
                if (mines.includes(index)) {
                    gameOver(false, index);
                } else {
                    SoundManager.sfxDig(); // KAZMA SESƒ∞
                    reveal(index);
                }
            }
        }

        function handleRightClick(index, cell) {
            if (!gameActive || gridState[index] === 1) return;
            toggleFlag(index, cell);
        }

        function toggleFlag(index, cell) {
            SoundManager.sfxFlag(); // BAYRAK SESƒ∞
            if (gridState[index] === 0) {
                gridState[index] = 2;
                cell.classList.add('flagged');
            } else if (gridState[index] === 2) {
                gridState[index] = 0;
                cell.classList.remove('flagged');
            }
            updateCounter();
        }

        function setMode(mode) {
            SoundManager.sfxClick();
            currentMode = mode;
            document.getElementById('btn-dig').classList.toggle('selected', mode === 'dig');
            document.getElementById('btn-flag').classList.toggle('selected', mode === 'flag');
        }

        function reveal(index) {
            const cell = document.querySelector(`.cell[data-index='${index}']`);
            if (!cell || gridState[index] !== 0) return;

            gridState[index] = 1;
            revealedCount++;
            cell.classList.add('revealed');
            cell.classList.remove('flagged');

            const count = countMines(index);
            if (count > 0) {
                cell.innerText = count;
                cell.classList.add(`val-${count}`);
            } else {
                getNeighbors(index).forEach(n => reveal(n));
            }

            if (revealedCount === totalCells - mines.length) {
                gameOver(true);
            }
        }

        function countMines(index) {
            return getNeighbors(index).filter(n => mines.includes(n)).length;
        }

        function getNeighbors(index) {
            const r = Math.floor(index / gridSize);
            const c = index % gridSize;
            const neighbors = [];
            for(let dr=-1; dr<=1; dr++){
                for(let dc=-1; dc<=1; dc++){
                    if(dr===0 && dc===0) continue;
                    const nr=r+dr, nc=c+dc;
                    if(nr>=0 && nr<gridSize && nc>=0 && nc<gridSize) neighbors.push(nr*gridSize+nc);
                }
            }
            return neighbors;
        }

        // --- Oyun Sonu ---
        function gameOver(win, explodedIndex) {
            gameActive = false;
            clearInterval(timerInterval);

            if (win) {
                SoundManager.sfxWin(); // KAZANMA SESƒ∞
                document.getElementById('face-btn').innerText = "üòé";
                mines.forEach(m => {
                    const c = document.querySelector(`.cell[data-index='${m}']`);
                    if(c && !c.classList.contains('flagged')) c.classList.add('flagged');
                });
                setTimeout(() => alert("TEBRƒ∞KLER! KAZANDINIZ!"), 100);
            } else {
                // KAYBETME
                SoundManager.sfxExplode(); // PATLAMA SESƒ∞
                
                // Ekranƒ± salla
                document.body.classList.add('shake-screen');
                document.getElementById('face-btn').innerText = "üòµ";
                
                mines.forEach(m => {
                    const c = document.querySelector(`.cell[data-index='${m}']`);
                    if(!c) return;

                    if (m === explodedIndex) {
                        c.classList.add('exploded');
                    } else if (!c.classList.contains('flagged')) {
                        c.classList.add('mine-reveal');
                    }
                });

                gridState.forEach((s, i) => {
                    if (s === 2 && !mines.includes(i)) {
                        const c = document.querySelector(`.cell[data-index='${i}']`);
                        c.classList.add('false-flag');
                        c.classList.remove('flagged');
                    }
                });
            }
        }

        function updateCounter() {
            let val = isSettingPhase ? (maxMines - mines.length) : (mines.length - gridState.filter(s=>s===2).length);
            document.getElementById('mine-display').innerText = Math.max(0, val).toString().padStart(3,'0');
        }

        function resetGame() {
            SoundManager.sfxClick();
            clearInterval(timerInterval);
            switchScreen('screen-setup');
            document.getElementById('timer-display').innerText = "000";
            document.getElementById('face-btn').innerText = "üôÇ";
            document.body.classList.remove('shake-screen');
        }

        function switchScreen(id) {
