    <script>
        // --- Global Deƒüi≈ükenler ---
        let gridSize = 8;
        let totalCells = 64;
        let maxMines = 0;
        let mines = []; // Mayƒ±nlarƒ±n indekslerini tutar
        let gridState = []; // 0:kapalƒ±, 1:a√ßƒ±k, 2:bayrak
        let gameActive = false;
        let isSettingPhase = true;
        let currentMode = 'dig';
        let revealedCount = 0;
        let timer = 0;
        let timerInterval;

        // --- Ayarlar ---
        function selectSize(n) {
            gridSize = n;
            totalCells = n * n;
            document.querySelectorAll('#screen-setup .classic-btn').forEach(b => b.classList.remove('selected'));
            event.target.classList.add('selected');
        }

        // --- Faz 1: Tuzak Kurma Ba≈ülangƒ±cƒ± ---
        function startGame() {
            isSettingPhase = true;
            mines = [];
            maxMines = Math.floor(totalCells * 0.20); // %20 kuralƒ±
            
            document.getElementById('screen-setup').classList.remove('active');
            document.getElementById('screen-game').classList.add('active');
            
            // Grid'i P1 moduna al (Kƒ±rmƒ±zƒ± g√∂r√ºn√ºm i√ßin)
            document.getElementById('game-grid').classList.add('p1-view');

            // UI Ayarla
            document.getElementById('game-status-text').innerText = "OYUNCU 1: TUZAK KURMA";
            document.getElementById('p1-controls').style.display = 'block';
            document.getElementById('p2-controls').style.display = 'none';
            document.getElementById('max-lbl').innerText = maxMines;
            document.getElementById('face-btn').innerText = "üôÇ";
            
            updateCounter();
            createGrid(handleP1Click);
        }

        // --- Grid Olu≈üturma (Optimize Edilmi≈ü) ---
        function createGrid(clickHandler) {
            const grid = document.getElementById('game-grid');
            grid.innerHTML = ''; // Temizle
            grid.style.gridTemplateColumns = `repeat(${gridSize}, 1fr)`;
            
            // H√ºcre boyutunu responsive yap
            const availableWidth = Math.min(window.innerWidth * 0.90, 400); 
            // 8px frame padding + borderlar i√ßin ufak tolerans
            
            for (let i = 0; i < totalCells; i++) {
                const cell = document.createElement('div');
                cell.classList.add('cell');
                cell.dataset.index = i;
                
                // P1 modundaysak ve mayƒ±n varsa, hemen g√∂rsel sƒ±nƒ±fƒ± ekle
                // (P2 moduna ge√ßerken grid yeniden yaratƒ±lacaƒüƒ± i√ßin bu sƒ±nƒ±f P2'de olmayacak)
                if (isSettingPhase && mines.includes(i)) {
                    cell.classList.add('has-mine');
                }

                // Efektler
                addPressEffects(cell);
                
                // Tƒ±klama Logic
                cell.onclick = () => clickHandler(i, cell);
                cell.oncontextmenu = (e) => {
                    e.preventDefault();
                    if (!isSettingPhase) handleRightClick(i, cell);
                };

                grid.appendChild(cell);
            }
        }

        function addPressEffects(cell) {
            cell.addEventListener('mousedown', () => {
                if(!isSettingPhase && gameActive) document.getElementById('face-btn').innerText = "üòÆ";
            });
            cell.addEventListener('touchstart', () => { // Mobil i√ßin
                if(!isSettingPhase && gameActive) document.getElementById('face-btn').innerText = "üòÆ";
            }, {passive: true});

            const resetFace = () => {
                if(!isSettingPhase && gameActive) document.getElementById('face-btn').innerText = "üôÇ";
            };
            cell.addEventListener('mouseup', resetFace);
            cell.addEventListener('touchend', resetFace);
        }

        // --- Faz 1 Mantƒ±ƒüƒ± (P1) ---
        function handleP1Click(index, cell) {
            const mineIndex = mines.indexOf(index);
            
            // Titre≈üim efekti (mobil)
            if (navigator.vibrate) navigator.vibrate(20);

            if (mineIndex > -1) {
                // Mayƒ±n kaldƒ±r
                mines.splice(mineIndex, 1);
                cell.classList.remove('has-mine');
            } else {
                // Mayƒ±n koy
                if (mines.length < maxMines) {
                    mines.push(index);
                    cell.classList.add('has-mine');
                } else {
                    // Dolu uyarƒ±sƒ± efekti
                    cell.style.backgroundColor = "white";
                    setTimeout(() => cell.style.backgroundColor = "", 100);
                }
            }
            updateCounter();
        }

        function finishSetting() {
            if (mines.length < 3) {
                alert("Oyunun zevkli olmasƒ± i√ßin en az 3 mayƒ±n yerle≈ütirmelisin!");
                return;
            }
            document.getElementById('screen-game').classList.remove('active');
            document.getElementById('screen-handover').classList.add('active');
        }

        // --- Faz 2 & 3: √á√∂zme Ba≈ülangƒ±cƒ± (P2) ---
        function startSolving() {
            isSettingPhase = false;
            gameActive = true;
            gridState = new Array(totalCells).fill(0);
            revealedCount = 0;
            currentMode = 'dig';
            
            // P1 g√∂r√ºn√ºm sƒ±nƒ±fƒ±nƒ± kaldƒ±r (Kritik G√ºvenlik)
            document.getElementById('game-grid').classList.remove('p1-view');

            // Saya√ßlarƒ± ba≈ülat
            timer = 0;
            clearInterval(timerInterval);
            timerInterval = setInterval(() => {
                timer++;
                if(timer < 999) document.getElementById('timer-display').innerText = timer.toString().padStart(3, '0');
            }, 1000);

            document.getElementById('screen-handover').classList.remove('active');
            document.getElementById('screen-game').classList.add('active');

            // UI G√ºncelle
            document.getElementById('game-status-text').innerText = "OYUNCU 2: √á√ñZ√úC√ú";
            document.getElementById('p1-controls').style.display = 'none';
            document.getElementById('p2-controls').style.display = 'flex';
            document.getElementById('face-btn').innerText = "üôÇ";
            
            updateCounter(); 
            
            // Grid'i YENƒ∞DEN OLU≈ûTUR (Mayƒ±n sƒ±nƒ±flarƒ± olmadan)
            // Bu adƒ±m Oyuncu 2'nin kaynak koduna baksa bile mayƒ±nlarƒ± g√∂rememesini saƒülar.
            createGrid(handleP2Click); 
        }

        // --- Faz 3 Mantƒ±ƒüƒ± (P2) ---
        function handleP2Click(index, cell) {
            if (!gameActive || gridState[index] === 1) return;

            if (currentMode === 'flag') {
                toggleFlag(index, cell);
            } else {
                if (gridState[index] === 2) return; // Bayraklƒ±ysa a√ßma
                
                if (mines.includes(index)) {
                    // G√úM!
                    gameOver(false, index);
                } else {
                    // A√ß
                    reveal(index);
                }
            }
        }

        function handleRightClick(index, cell) {
            if (!gameActive || gridState[index] === 1) return;
            toggleFlag(index, cell);
        }

        function toggleFlag(index, cell) {
            if (navigator.vibrate) navigator.vibrate(20);
            
            if (gridState[index] === 0) {
                gridState[index] = 2;
                cell.classList.add('flagged');
            } else if (gridState[index] === 2) {
                gridState[index] = 0;
                cell.classList.remove('flagged');
            }
            updateCounter();
        }

        function setMode(mode) {
            currentMode = mode;
            document.getElementById('btn-dig').classList.toggle('selected', mode === 'dig');
            document.getElementById('btn-flag').classList.toggle('selected', mode === 'flag');
        }

        function reveal(index) {
            // DOM √ºzerinden h√ºcreyi bul (index ile)
            const cell = document.querySelector(`.cell[data-index='${index}']`);
            
            if (!cell || gridState[index] !== 0) return;

            gridState[index] = 1; // A√ßƒ±ldƒ± i≈üaretle
            revealedCount++;
            cell.classList.add('revealed');
            cell.classList.remove('flagged'); // Yanlƒ±≈ülƒ±kla bayrak konduysa sil

            // Kom≈üu mayƒ±n sayƒ±sƒ±
            const count = countMines(index);
            if (count > 0) {
                cell.innerText = count;
                cell.classList.add(`val-${count}`);
            } else {
                // Bo≈üsa (0), kom≈üularƒ± da otomatik a√ß (Flood Fill)
                getNeighbors(index).forEach(n => reveal(n));
            }

            checkWin();
        }

        function countMines(index) {
            return getNeighbors(index).filter(n => mines.includes(n)).length;
        }

        function getNeighbors(index) {
            const r = Math.floor(index / gridSize);
            const c = index % gridSize;
            const neighbors = [];
            for(let dr=-1; dr<=1; dr++){
                for(let dc=-1; dc<=1; dc++){
                    if(dr===0 && dc===0) continue;
                    const nr = r+dr, nc = c+dc;
                    if(nr>=0 && nr<gridSize && nc>=0 && nc<gridSize) {
                        neighbors.push(nr*gridSize + nc);
                    }
                }
            }
            return neighbors;
        }

        function checkWin() {
            if (revealedCount === totalCells - mines.length) {
                gameOver(true);
            }
        }

        function gameOver(win, explodedIndex) {
            gameActive = false;
            clearInterval(timerInterval);
            
            if (navigator.vibrate) navigator.vibrate(win ? [100,50,100] : [500]);

            if (win) {
                document.getElementById('face-btn').innerText = "üòé";
                document.getElementById('game-status-text').innerText = "TEBRƒ∞KLER! OYUNCU 2 KAZANDI";
                // Kalan mayƒ±nlarƒ± bayrakla
                mines.forEach(m => {
                    const c = document.querySelector(`.cell[data-index='${m}']`);
                    if(c && !c.classList.contains('flagged')) {
                        c.classList.add('flagged');
                    }
                });
            } else {
                document.getElementById('face-btn').innerText = "üòµ";
                document.getElementById('game-status-text').innerText = "PATLADIN! OYUNCU 1 KAZANDI";
                
                // T√ºm mayƒ±nlarƒ± g√∂ster
                mines.forEach(m => {
                    const c = document.querySelector(`.cell[data-index='${m}']`);
                    if (!c) return;

                    if (m === explodedIndex) {
                        c.classList.add('exploded'); // Patlayan mayƒ±n (kƒ±rmƒ±zƒ±)
                    } else if (!c.classList.contains('flagged')) {
                        // Bayrak konmamƒ±≈ü mayƒ±nlarƒ± a√ß
                        c.classList.add('revealed'); 
                        c.style.backgroundColor = "#c0c0c0";
                        // Dinamik yaratƒ±lan bir element ekleyelim
                        c.innerHTML = "üí£"; 
                    }
                });

                // Hatalƒ± bayraklarƒ± g√∂ster (X i≈üareti)
                gridState.forEach((state, idx) => {
                    if (state === 2 && !mines.includes(idx)) {
                        const c = document.querySelector(`.cell[data-index='${idx}']`);
                        c.style.backgroundColor = "#ffcccc";
                        c.innerHTML = "‚ùå";
                    }
                });
            }
        }

        function updateCounter() {
            let count;
            if (isSettingPhase) {
                // P1: Ka√ß tane daha koyabilirim? veya Ka√ß tane koydum?
                // Klasik mantƒ±kta kalan mayƒ±n g√∂sterilir.
                count = maxMines - mines.length;
                document.getElementById('max-lbl').innerText = count; // Text g√ºncelle
